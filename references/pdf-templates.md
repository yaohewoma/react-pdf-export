# PDF 模板模块

## 概述

提供可复用的 PDF 模板系统，支持自定义页眉、页脚、Logo 和布局。

## 配置

## 实现

```typescript
import { jsPDF } from 'jspdf';

/**
 * PDF 模板管理器
 * 提供可复用的页面模板，支持页眉、页脚和自定义布局
 */
export class PDFTemplateManager {
  private config: PDFTemplateConfig;
  private pdf: jsPDF;
  private currentPage: number = 0;

  constructor(config: Partial<PDFTemplateConfig> = {}) {
    this.config = { ...DEFAULT_TEMPLATE_CONFIG, ...config };
    this.pdf = new jsPDF({
      orientation: this.config.orientation,
      unit: 'mm',
      format: this.config.pageSize
    });
  }

  /**
   * 获取 PDF 实例
   */
  getPdf(): jsPDF {
    return this.pdf;
  }

  /**
   * 添加新页面
   */
  addPage(): void {
    this.currentPage++;
    if (this.currentPage > 1) {
      this.pdf.addPage();
    }
    this.renderHeader();
    this.renderFooter();
  }

  /**
   * 渲染页眉
   */
  private renderHeader(): void {
    const { header, margins } = this.config;
    if (!header?.enabled) return;

    const pageWidth = this.pdf.internal.pageSize.getWidth();
    const headerY = margins.top - header.height;

    // 绘制 Logo
    if (header.logo?.url) {
      const logoX = this.getAlignedX(
        header.logo.position,
        pageWidth,
        header.logo.width,
        margins
      );
      this.pdf.addImage(
        header.logo.url,
        'PNG',
        logoX,
        headerY,
        header.logo.width,
        header.logo.height
      );
    }

    // 绘制标题
    if (header.title?.text) {
      this.pdf.setFontSize(header.title.fontSize);
      this.pdf.setTextColor(header.title.fontColor);
      
      const titleX = this.getAlignedX(
        header.title.position,
        pageWidth,
        this.pdf.getTextWidth(header.title.text),
        margins
      );
      
      this.pdf.text(
        header.title.text,
        titleX,
        headerY + header.height / 2
      );
    }

    // 绘制分隔线
    if (header.line?.enabled) {
      this.pdf.setDrawColor(header.line.color);
      this.pdf.setLineWidth(header.line.width);
      this.pdf.line(
        margins.left,
        headerY + header.height,
        pageWidth - margins.right,
        headerY + header.height
      );
    }
  }

  /**
   * 渲染页脚
   */
  private renderFooter(): void {
    const { footer, margins } = this.config;
    if (!footer?.enabled) return;

    const pageWidth = this.pdf.internal.pageSize.getWidth();
    const pageHeight = this.pdf.internal.pageSize.getHeight();
    const footerY = pageHeight - margins.bottom;

    // 绘制分隔线
    if (footer.line?.enabled) {
      this.pdf.setDrawColor(footer.line.color);
      this.pdf.setLineWidth(footer.line.width);
      this.pdf.line(
        margins.left,
        footerY,
        pageWidth - margins.right,
        footerY
      );
    }

    // 绘制页码
    if (footer.pageNumber?.enabled) {
      const totalPages = this.pdf.getNumberOfPages();
      let pageText = '';
      
      switch (footer.pageNumber.format) {
        case 'page':
          pageText = `${this.currentPage}`;
          break;
        case 'page-total':
          pageText = `${this.currentPage} / ${totalPages}`;
          break;
        case 'custom':
          pageText = footer.pageNumber.customFormat
            ?.replace('{page}', String(this.currentPage))
            ?.replace('{total}', String(totalPages)) || '';
          break;
      }

      this.pdf.setFontSize(10);
      this.pdf.setTextColor('#666666');
      
      const pageX = this.getAlignedX(
        footer.pageNumber.position,
        pageWidth,
        this.pdf.getTextWidth(pageText),
        margins
      );
      
      this.pdf.text(pageText, pageX, footerY + 5);
    }

    // 绘制文本
    if (footer.text?.text) {
      this.pdf.setFontSize(footer.text.fontSize);
      this.pdf.setTextColor(footer.text.fontColor);
      
      const textX = this.getAlignedX(
        footer.text.position,
        pageWidth,
        this.pdf.getTextWidth(footer.text.text),
        margins
      );
      
      this.pdf.text(footer.text.text, textX, footerY + 5);
    }
  }

  /**
   * 获取对齐后的 X 坐标
   */
  private getAlignedX(
    position: 'left' | 'center' | 'right',
    pageWidth: number,
    elementWidth: number,
    margins: { left: number; right: number }
  ): number {
    switch (position) {
      case 'left':
        return margins.left;
      case 'center':
        return (pageWidth - elementWidth) / 2;
      case 'right':
        return pageWidth - margins.right - elementWidth;
      default:
        return margins.left;
    }
  }

  /**
   * 获取内容区域
   */
  getContentArea(): { x: number; y: number; width: number; height: number } {
    const { margins, header, footer } = this.config;
    const pageWidth = this.pdf.internal.pageSize.getWidth();
    const pageHeight = this.pdf.internal.pageSize.getHeight();

    return {
      x: margins.left,
      y: margins.top + (header?.enabled ? header.height : 0),
      width: pageWidth - margins.left - margins.right,
      height: pageHeight - margins.top - margins.bottom - 
              (header?.enabled ? header.height : 0) - 
              (footer?.enabled ? footer.height : 0)
    };
  }

  /**
   * 添加文本内容
   */
  addText(text: string, options: {
    x?: number;
    y?: number;
    fontSize?: number;
    fontColor?: string;
    maxWidth?: number;
    align?: 'left' | 'center' | 'right';
  } = {}): void {
    const area = this.getContentArea();
    const {
      x = area.x,
      y = area.y,
      fontSize = 12,
      fontColor = '#333333',
      maxWidth = area.width,
      align = 'left'
    } = options;

    this.pdf.setFontSize(fontSize);
    this.pdf.setTextColor(fontColor);
    this.pdf.text(text, x, y, { maxWidth, align });
  }

  /**
   * 添加表格
   */
  addTable(data: string[][], options: {
    x?: number;
    y?: number;
    cellWidth?: number;
    cellHeight?: number;
    headerColor?: string;
    headerTextColor?: string;
    borderColor?: string;
  } = {}): void {
    const area = this.getContentArea();
    const {
      x = area.x,
      y = area.y,
      cellWidth = area.width / data[0].length,
      cellHeight = 10,
      headerColor = '#f5f5f5',
      headerTextColor = '#333333',
      borderColor = '#e5e5e5'
    } = options;

    data.forEach((row, rowIndex) => {
      row.forEach((cell, colIndex) => {
        const cellX = x + colIndex * cellWidth;
        const cellY = y + rowIndex * cellHeight;

        // 绘制背景
        if (rowIndex === 0) {
          this.pdf.setFillColor(headerColor);
          this.pdf.rect(cellX, cellY, cellWidth, cellHeight, 'F');
        }

        // 绘制边框
        this.pdf.setDrawColor(borderColor);
        this.pdf.rect(cellX, cellY, cellWidth, cellHeight);

        // 绘制文字
        this.pdf.setFontSize(10);
        this.pdf.setTextColor(rowIndex === 0 ? headerTextColor : '#333333');
        this.pdf.text(cell, cellX + 2, cellY + cellHeight / 2 + 1);
      });
    });
  }

  /**
   * 添加图片
   */
  addImage(imageUrl: string, options: {
    x?: number;
    y?: number;
    width?: number;
    height?: number;
  } = {}): void {
    const area = this.getContentArea();
    const {
      x = area.x,
      y = area.y,
      width = area.width,
      height = area.height
    } = options;

    this.pdf.addImage(imageUrl, 'PNG', x, y, width, height);
  }

  /**
   * 保存 PDF
   */
  save(fileName: string = 'document.pdf'): void {
    this.pdf.save(fileName);
  }
}
```

## 使用示例

```typescript
import { PDFTemplateManager, ENTERPRISE_TEMPLATE } from './pdf-templates';

// 创建模板管理器
const template = new PDFTemplateManager(ENTERPRISE_TEMPLATE);

// 添加页面
template.addPage();

// 添加标题
template.addText('数据分析报告', {
  fontSize: 24,
  fontColor: '#1a1a1a',
  align: 'center'
});

// 添加表格
template.addTable([
  ['指标', '数值', '趋势'],
  ['总用户数', '10,000', '↑ 15%'],
  ['活跃用户', '6,500', '↑ 8%'],
  ['转化率', '65%', '↓ 2%']
], { y: 50 });

// 保存
template.save('report.pdf');
```

## 注意事项

1. **字体支持**：中文字体需要额外加载
2. **图片预加载**：Logo 等图片需要提前加载
3. **页面溢出**：内容超出页面时需要自动分页
4. **样式一致性**：确保模板在不同设备上显示一致
